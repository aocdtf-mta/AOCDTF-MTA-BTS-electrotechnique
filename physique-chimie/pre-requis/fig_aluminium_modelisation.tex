	\def\proton(#1,#2){\shade[ball color=yellow, opacity=.8] ({rnd*#1},{rnd*#2}) circle (7pt) node {\texttt{+}};}
	\def\neutron(#1,#2){\shade[ball color=orange, opacity=.8] ({rnd*#1},{rnd*#2}) circle (7pt) node {\texttt{0}};}
	\def\electron{\shade[ball color=gray!30] (0,0) circle (7pt); \node at (0,0) {\texttt{\color{black}-}};}

	\def\ones2orbit(#1,#2,#3){
 		\draw[
 	 		color=green,
  			rotate=#1,
 	 		postaction=decorate,
 	 		decoration={markings,pre=moveto,pre length=#3,
 			 mark=at position #2 with {\electron},}]
  		(0,0) ellipse (0.9 and 1.8);}
